'use strict';
 
// Represents a User
function UserModel(userData) {
  if (userData) {
    this.id = userData.id;
    this.name = userData.name;
    this.setEmail(userData.email);
    this.setFavoriteColor(userData.favoriteColor);
  }
}

UserModel.prototype.setEmail = function(email) {
  if (this.emailRegex.test(email)) {
    this.email = email;
  } else {
    throw new Error('bad email format: ' + email);
  }
};
 
UserModel.prototype.setFavoriteColor = function(color) {
  if (color && color.match(this.colorRegex)) {
    this.favoriteColor = color;
  } else {
    throw new Error('bad color: ' + color + ' try ' + UserModel.prototype.colorRegex);
  }
};
